// 計算直線與圓相交的座標
int lineCircleIntersection(int x1, int y1, int x2, int y2, int cx, int cy, int cr, int *x, int *y) { 
    float dx, dy, A , B、C、檢測、t；
 
    dx = x2 - x1; 
    dy = y2 - y1；
 
    A = dx * dx + dy * dy; 
    B = 2 * (dx * (x1 - cx) + dy * (y1 - cy)); 
    C = (x1 - cx) * (x1 - cx) + (y1 - cy) * (y1 - cy) - cr * cr; 
 
    det = B * B - 4 * A * C；
 
    if ((A <= 0.00001) || (det < 0)) { 
        // 沒有實數解
        return 0; 
    } 
 
    t = (-B - sqrt(det)) / (2 * A); 
    *x = x1 + t * dx; 
    *y = y1 + t * dy; 
 
    返回1；
}
